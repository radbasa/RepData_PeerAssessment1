  1. Create a new dataset based on the original with the missing data filled with the daily average for the 5-minute interval
test <- function(s,t) {
    if ( is.na(s) )
      return( dailyAverage$steps[ dailyAverage$interval == t ] )
    else
      return( s )
  }
